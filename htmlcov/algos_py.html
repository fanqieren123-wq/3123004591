3def lcs_length(a: str, b: str) -> int: 

4 """LCS 长度（滚动数组，O(min(n,m)) 空间）""" 

5 m, n = len(a), len(b) 

6 if m == 0 or n == 0: 

7 return 0 

8 # 让 b 成更短的一边，减少列数 

9 if n > m: 

10 a, b = b, a 

11 m, n = n, m 

12 

13 prev = [0] * (n + 1) 

14 curr = [0] * (n + 1) 

15 for i in range(1, m + 1): 

16 ai = a[i - 1] 

17 for j in range(1, n + 1): 

18 if ai == b[j - 1]: 

19 curr[j] = prev[j - 1] + 1 

20 else: 

21 left = curr[j - 1] 

22 up = prev[j] 

23 curr[j] = left if left >= up else up 

24 prev, curr = curr, [0] * (n + 1) 

25 return prev[n]

28 """返回 LCS 相似度百分比（以“抄袭文本长度”为分母）。copied 为空则 0.0。"""
